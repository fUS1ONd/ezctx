class GroqKeyPool extends ChangeNotifier {
  GroqKeyPool({List<String> initialKeys = const []});

  int get aliveKeyCount;
  List<String> get allKeys;                       // все ключи включая заблокированные

  Future<String> acquireKey();                    // async, round-robin, ждёт живого
  void reportRateLimited(String key, int retryAfterSeconds);
  void releaseKey(String key);                    // no-op в текущей реализации
  List<KeyStatus> getStatuses();
  KeyStatus getStatusForKey(String rawKey);
  void addKey(String raw);
  void removeKey(String raw);
}

// lib/core/error/app_exception.dart
class RateLimitException extends AppException {
  final int retryAfterSeconds;
  const RateLimitException(super.message, {this.retryAfterSeconds = 60});
}

class AllKeysBlockedException extends AppException {
  const AllKeysBlockedException([super.message = 'Все ключи заблокированы. Ожидание…']);
}

// lib/core/constants/app_constants.dart
class AppConstants {
  static const int kMaxConcurrentChunks = 5;
}
