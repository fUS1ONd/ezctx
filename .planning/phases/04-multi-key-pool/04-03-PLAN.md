      import 'dart:async';
      import 'package:flutter/material.dart';
      import '../../core/constants/design_tokens.dart';
      import '../../features/transcription/groq_key_pool.dart';

    Класс KeyStatusTile extends StatefulWidget:
      final KeyStatus status;
      const KeyStatusTile({super.key, required this.status});

    Класс _KeyStatusTileState:
      Timer? _timer;

      initState():
        - if (widget.status is BlockedKeyStatus) → стартовать Timer.periodic(1s, (_) { if (mounted) setState((){}); })

      didUpdateWidget(KeyStatusTile oldWidget):
        - if (widget.status.runtimeType != oldWidget.status.runtimeType):
            _timer?.cancel(); _timer = null;
            if (widget.status is BlockedKeyStatus) → стартовать Timer.periodic(...)

      dispose():
        _timer?.cancel();   // ОБЯЗАТЕЛЬНО — Pitfall 3 из RESEARCH.md
        super.dispose();

      build():
        final s = widget.status;
        if (s is BlockedKeyStatus):
          remaining = s.blockedUntil.difference(DateTime.now())
          if remaining.isNegative → return _activeBadge()
          return _blockedBadge(_formatRemaining(remaining))
        return _activeBadge()

      _formatRemaining(Duration d) → String:
        h = d.inHours.toString().padLeft(2, '0')
        m = (d.inMinutes % 60).toString().padLeft(2, '0')
        s = (d.inSeconds % 60).toString().padLeft(2, '0')
        return '$h:$m:$s'

      _activeBadge():
        Row(mainAxisSize: MainAxisSize.min, children: [
          Icon(Icons.circle, size: 8, color: AppColors.good),
          SizedBox(width: AppSpacing.xs),
          Text('Активен', style: AppTextStyles.label),
        ])

      _blockedBadge(String countdown):
        Row(mainAxisSize: MainAxisSize.min, children: [
          Icon(Icons.circle, size: 8, color: AppColors.bad),
          SizedBox(width: AppSpacing.xs),
          Text('До $countdown', style: AppTextStyles.label.copyWith(color: AppColors.bad)),
        ])
