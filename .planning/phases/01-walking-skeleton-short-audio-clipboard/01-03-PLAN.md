    import '../../core/constants/app_constants.dart';

    /// Результат валидации файла. Может быть success или одна из ошибок.
    sealed class FileValidationResult {
      const FileValidationResult();

      const factory FileValidationResult.ok() = _Ok;
      const factory FileValidationResult.invalidExtension(String message) = _InvalidExtension;
      const factory FileValidationResult.tooLarge(String message) = _TooLarge;

      bool get isOk => this is _Ok;
      String? get errorMessage => switch (this) {
        _Ok() => null,
        _InvalidExtension(message: final m) => m,
        _TooLarge(message: final m) => m,
      };
    }
    class _Ok extends FileValidationResult { const _Ok(); }
    class _InvalidExtension extends FileValidationResult {
      final String message;
      const _InvalidExtension(this.message);
    }
    class _TooLarge extends FileValidationResult {
      final String message;
      const _TooLarge(this.message);
    }

    /// Чистая валидация: whitelist расширений + лимит размера.
    /// Никакого I/O, никакого file_picker — только проверка переданных значений.
    class FileValidator {
      const FileValidator();

      static const _invalidExtensionMessage =
          'Формат файла не поддерживается. Выберите mp3, wav, m4a, ogg или flac.';
      static const _tooLargeMessage =
          'Файл слишком большой (максимум 19 МБ). Поддержка больших файлов — в следующем обновлении.';

      FileValidationResult validate({
        required String path,
        required int sizeBytes,
      }) {
        final ext = _extractExtension(path);
        if (ext == null ||
            !AppConstants.supportedAudioExtensions.contains(ext)) {
          return const FileValidationResult.invalidExtension(_invalidExtensionMessage);
        }
        if (sizeBytes > AppConstants.maxFileSizeBytes) {
          return const FileValidationResult.tooLarge(_tooLargeMessage);
        }
        return const FileValidationResult.ok();
      }

      /// Извлекает расширение в нижнем регистре без точки.
      /// 'song.MP3' → 'mp3', '/path/to/file.tar.gz' → 'gz', 'noext' → null.
      static String? _extractExtension(String path) {
        final lastDot = path.lastIndexOf('.');
        if (lastDot == -1 || lastDot == path.length - 1) return null;
        // Учесть что точка может быть в имени папки.
        final lastSep = path.lastIndexOf(RegExp(r'[/\\]'));
        if (lastDot < lastSep) return null;
        return path.substring(lastDot + 1).toLowerCase();
      }
    }

    ```dart
    import 'package:file_picker/file_picker.dart';

    import '../../core/constants/app_constants.dart';
    import '../../core/error/app_exception.dart';
    import 'file_validator.dart';
    import 'selected_audio_file.dart';

    /// Результат попытки выбора файла:
    /// - `picked` с выбранным валидным файлом;
    /// - `cancelled` если пользователь отменил диалог;
    /// - throw ValidationException если файл не прошёл валидацию.
    sealed class FilePickResult {
      const FilePickResult();
      const factory FilePickResult.picked(SelectedAudioFile file) = _Picked;
      const factory FilePickResult.cancelled() = _Cancelled;
    }
    class _Picked extends FilePickResult {
      final SelectedAudioFile file;
      const _Picked(this.file);
    }
    class _Cancelled extends FilePickResult {
      const _Cancelled();
    }

    /// Обёртка над системным file_picker.
    /// Применяет whitelist расширений и валидацию размера.
    class FilePickerService {
      const FilePickerService({
        FileValidator validator = const FileValidator(),
      }) : _validator = validator;

      final FileValidator _validator;

      /// Открывает системный диалог выбора файла.
      /// Возвращает [FilePickResult.picked] с валидным файлом
      /// или [FilePickResult.cancelled] если пользователь закрыл диалог.
      /// Бросает [ValidationException] если выбранный файл не проходит валидацию.
      Future<FilePickResult> pickAudioFile() async {
        final FilePickerResult? result = await FilePicker.platform.pickFiles(
          type: FileType.custom,
          allowedExtensions: AppConstants.supportedAudioExtensions.toList(),
          allowMultiple: false,
          withReadStream: false,
          withData: false,
        );

        if (result == null || result.files.isEmpty) {
          return const FilePickResult.cancelled();
        }

        final PlatformFile platformFile = result.files.first;
        final String? path = platformFile.path;

        if (path == null) {
          // Pitfall 6: Scoped Storage на Android 10+ может вернуть path=null
          // для некоторых SAF-источников (cloud, content providers).
          throw const ValidationException(
            'Не удалось получить путь к файлу. Попробуйте выбрать файл из «Загрузки» или «Музыка».',
          );
        }

        final validation = _validator.validate(
          path: path,
          sizeBytes: platformFile.size,
        );

        if (!validation.isOk) {
          throw ValidationException(validation.errorMessage!);
        }

        // Извлечь имя и расширение для модели.
        final name = platformFile.name;
        final ext = platformFile.extension?.toLowerCase() ??
            _extractExtensionFallback(name);

        return FilePickResult.picked(
          SelectedAudioFile(
            path: path,
            name: name,
            sizeBytes: platformFile.size,
            extension: ext ?? '',
          ),
        );
      }

      static String? _extractExtensionFallback(String name) {
        final i = name.lastIndexOf('.');
        if (i == -1 || i == name.length - 1) return null;
        return name.substring(i + 1).toLowerCase();
      }
    }
    ```
